    def confirm(self, sig: Literal[-1, 1]) -> ActionNode:
        self.signal = sig

        if sig > 0:
            return LongAction(sig=sig)
        elif sig < 0:
            return ShortAction(sig=sig)

        if not LGM.inspection_mode:
            LOGGER.warning(f'{self} received a confirmation of {sig=}! Which is not expected.')

        return NoAction()
